Lua интеграция
Встроенный ИИ отвечает текстом, а твой движок на Lua может вызывать HTTP‑запросы к тому же API. 
-- пример Lua-запроса (Luasocket/http)
local json = require("json")
local http = require("socket.http")

local body = json.encode({
  model = "claude-sonnet-4.6",
  max_tokens = 512,
  messages = {
    { role = "user", content = "Привет, Sonnet!" }
  }
})

local resp_body = {}
local ok, status = http.request{
  url = "https://api.anthropic.com/v1/messages",
  method = "POST",
  headers = {
    ["Content-Type"] = "application/json",
    ["x-api-key"] = os.getenv("ANTHROPIC_KEY")
  },
  source = ltn12.source.string(body),
  sink = ltn12.sink.table(resp_body)
}
